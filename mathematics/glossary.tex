\DeclareMathOperator*{\argmax}{arg\,max}
\DeclareMathOperator*{\argmin}{arg\,min}

% ---- Glossary Entries ----

\longnewglossaryentry{domain}{name=domain}
{%
  The set of ``input'' or argument variables for which a given function is defined.
}

\longnewglossaryentry{argmax}{name=\ensuremath{\argmax\limits_{x}f(x)}}
{%
  The value of $x$ for which $f(x)$ is maximal.
}


\longnewglossaryentry{argmin}{name=\ensuremath{\argmin\limits_{x}f(x)}}
{%
  The value of $x$ for which $f(x)$ is minimial.
}
